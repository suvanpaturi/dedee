# Build stage for model download
FROM ollama/ollama:latest as builder

# Install models during build
RUN ollama serve & \
    sleep 5 && \
    ollama pull llama2 && \
    ollama pull mistral && \
    ollama pull orca-mini && \
    mkdir -p /root/.ollama

# Final stage
FROM nvidia/cuda:12.2.0-base-ubuntu22.04

# Install required dependencies
RUN apt-get update && apt-get install -y \
    curl \
    wget \
    python3 \
    python3-pip \
    && rm -rf /var/lib/apt/lists/*

# Install Ollama
RUN curl -L https://ollama.ai/install.sh | sh

# Copy pre-downloaded models from builder
COPY --from=builder /root/.ollama /root/.ollama

# Install Python packages for inference script
RUN pip3 install fastapi uvicorn pydantic requests

# Copy inference service script
COPY inference_service.py /inference_service.py

# Expose API port
EXPOSE 8000

# Start inference service
CMD ["python3", "/inference_service.py"]
